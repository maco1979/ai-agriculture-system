# AI农业决策系统 - 智能体技能使用指南

## 🎯 概述

本文档介绍如何为AI农业决策系统中的各个智能体配置和使用技能。系统包含4个核心智能体，每个智能体都有专门的技能集。

## 🤖 智能体架构

### 1. 决策引擎智能体 (核心AI)
**职责**: 综合分析天气、作物、病虫害数据，提供农业决策建议
**位置**: `backend/src/decision_engine/`

### 2. 区块链智能体 (溯源)
**职责**: 农产品全生命周期数据上链和追溯
**位置**: `backend/src/blockchain/`

### 3. 边缘计算智能体 (设备控制)
**职责**: 物联网设备连接、数据采集和本地决策
**位置**: `backend/src/edge_computing/`

### 4. 用户交互智能体 (前端)
**职责**: 自然语言理解、个性化推荐和用户交互
**位置**: `frontend/src/agents/`

## 🔧 技能集成到智能体

### 决策引擎智能体集成示例

```python
# backend/src/decision_engine/agent.py
from skills.weather_skill import WeatherSkill
from skills.crop_skill import CropSkill
from skills.pest_skill import PestSkill

class DecisionEngineAgent:
    def __init__(self):
        self.weather_skill = WeatherSkill()
        self.crop_skill = CropSkill()
        self.pest_skill = PestSkill()
    
    async def make_decision(self, farm_data: dict) -> dict:
        """做出农业决策"""
        decisions = []
        
        # 1. 获取天气数据
        weather = await self.weather_skill.get_agriculture_weather(
            farm_data["location"]
        )
        
        # 2. 分析作物状态
        crop_status = self.crop_skill.calculate_growth_stage(
            farm_data["crop"],
            farm_data["planting_date"]
        )
        
        # 3. 评估病虫害风险
        pest_risk = self.pest_skill.predict_pest_risk(
            farm_data["crop"],
            weather["forecast"]["current"],
            crop_status["current_stage"]
        )
        
        # 4. 生成决策建议
        if pest_risk["high_risk_pests"]:
            decisions.append({
                "type": "病虫害防治",
                "priority": "高",
                "action": "立即采取防治措施",
                "details": pest_risk
            })
        
        if weather["agriculture_analysis"]["irrigation_recommendation"] == "降雨不足":
            decisions.append({
                "type": "灌溉",
                "priority": "中",
                "action": "建议灌溉",
                "details": weather["agriculture_analysis"]
            })
        
        return {
            "farm_id": farm_data["farm_id"],
            "timestamp": datetime.now().isoformat(),
            "decisions": decisions,
            "supporting_data": {
                "weather": weather,
                "crop_status": crop_status,
                "pest_risk": pest_risk
            }
        }
```

### 区块链智能体集成示例

```python
# backend/src/blockchain/agent.py
from skills.blockchain_skill import BlockchainSkill

class BlockchainAgent:
    def __init__(self):
        self.blockchain_skill = BlockchainSkill()
    
    async def record_agriculture_event(self, event_data: dict) -> dict:
        """记录农业事件到区块链"""
        event_type = event_data["type"]
        
        if event_type == "planting":
            return await self.blockchain_skill.record_planting(
                farm_id=event_data["farm_id"],
                crop_id=event_data["crop_id"],
                planting_data=event_data["data"]
            )
        
        elif event_type == "harvest":
            return await self.blockchain_skill.record_harvest(
                farm_id=event_data["farm_id"],
                crop_id=event_data["crop_id"],
                harvest_data=event_data["data"]
            )
        
        # ... 其他事件类型
    
    async def create_product_passport(self, product_id: str) -> dict:
        """创建产品数字护照"""
        trace_result = self.blockchain_skill.trace_product(product_id)
        verify_result = self.blockchain_skill.verify_product(product_id)
        
        return {
            "product_id": product_id,
            "passport_id": f"PASSPORT_{product_id}",
            "traceability": trace_result,
            "verification": verify_result,
            "qr_code_url": f"https://agri-blockchain.com/passport/{product_id}",
            "issue_date": datetime.now().isoformat()
        }
```

### 边缘计算智能体集成示例

```python
# backend/src/edge_computing/agent.py
from skills.device_skill import DeviceSkill

class EdgeComputingAgent:
    def __init__(self):
        self.device_skill = DeviceSkill()
        self.local_decisions = []
    
    async def monitor_devices(self):
        """监控设备状态"""
        devices = await self.device_skill.discover_devices()
        
        for device in devices:
            status = await self.device_skill.get_device_status(device["device_id"])
            
            if status["status"] == "故障":
                await self.handle_device_failure(device, status)
            
            elif device["type"] == "传感器":
                data = await self.device_skill.read_sensor_data(device["device_id"])
                await self.process_sensor_data(device, data)
    
    async def process_sensor_data(self, device: dict, data: dict):
        """处理传感器数据并做出本地决策"""
        # 本地决策：自动灌溉
        if (device["name"] == "土壤传感器" and 
            data["data"].get("soil_moisture", 100) < 25):
            
            decision = {
                "type": "自动灌溉",
                "device": "actuator_valve_001",
                "action": "open",
                "parameters": {"flow_rate": 8, "duration": 600},
                "reason": f"土壤湿度低: {data['data']['soil_moisture']}%"
            }
            
            # 执行决策
            result = await self.device_skill.control_device(
                device_id=decision["device"],
                command=decision["action"],
                parameters=decision["parameters"]
            )
            
            self.local_decisions.append({
                "decision": decision,
                "result": result,
                "timestamp": datetime.now().isoformat()
            })
```
